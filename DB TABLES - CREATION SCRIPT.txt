// PROCESS TABLE
CREATE TABLE TALENT4GIG.PROCESSES (
	ID INTEGER UNSIGNED auto_increment NOT NULL,
	NAME varchar(50) NOT NULL,
	DESCRIPTION varchar(200) NULL,
	PERIODICITY varchar(50) NOT NULL,
	CATEGORY varchar(25) NULL,
	CREATED DATETIME DEFAULT CURRENT_TIMESTAMP NULL,
	UPDATED TIMESTAMP NULL,
	CONSTRAINT PROCESSES_PK PRIMARY KEY (ID)
)
ENGINE=InnoDB
DEFAULT CHARSET=latin1
COLLATE=latin1_swedish_ci;
	

// EXECUTION TABLE
CREATE TABLE TALENT4GIG.EXECUTION_INFO (
	ID INTEGER UNSIGNED auto_increment NOT NULL,
	FID_PROCESS INTEGER UNSIGNED NOT NULL,
	START_TIME DATETIME NULL,
	END_TIME DATETIME NULL,
	FINAL_STATE ENUM('COMPLETED', 'COMPLETEDWITHERROR', 'NOTSTARTED') DEFAULT 'NOTSTARTED' NOT NULL;
	CONSTRAINT EXECUTION_INFO_PK PRIMARY KEY (ID)
)
ENGINE=InnoDB
DEFAULT CHARSET=latin1
COLLATE=latin1_swedish_ci;
